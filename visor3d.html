import os import json import base64 import requests from datetime import datetime, timedelta import time from io import StringIO # ============================================================ # FUNCIÓN PARA ENRIQUECER CON UBICACIÓN (CON POLÍGONOS Y AUTENTICACIÓN) # ============================================================ def enriquecer_con_ubicacion(puntos, regions_url, hf_token=None): """ Añade NAME_1 (región) y locality (localidad) a cada punto usando intersección espacial con polígonos """ if not puntos: return puntos try: import geopandas as gpd import pandas as pd except ImportError: print("[ADVERTENCIA] geopandas o pandas no instalados. No se enriquecerán los datos.") return puntos # 1. Crear GeoDataFrame con los puntos gdf_puntos = gpd.GeoDataFrame( pd.DataFrame([p['properties'] for p in puntos]), geometry=gpd.points_from_xy( [p['geometry']['coordinates'][0] for p in puntos], [p['geometry']['coordinates'][1] for p in puntos] ), crs='EPSG:4326' ) # 2. Cargar ubicaciones desde Hugging Face con autenticación try: print(f"[INFO] Cargando ubicaciones desde: {regions_url}") print(f"[INFO] El archivo es grande (70 MB), puede tardar unos segundos...") headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36" } if hf_token: headers["Authorization"] = f"Bearer {hf_token}" print("[INFO] ✅ Usando autenticación con token de Hugging Face") else: print("[ADVERTENCIA] ⚠️ No hay token HF_TOKEN. Si el Space es privado, fallará.") response = requests.get(regions_url, headers=headers, timeout=180) if response.status_code == 200: gdf_ubicaciones = gpd.read_file(StringIO(response.text)) print(f"[INFO] ✅ Ubicaciones cargadas: {len(gdf_ubicaciones)} polígonos") elif response.status_code == 401: print("[ERROR] ❌ Error 401: No autorizado. Verifica que HF_TOKEN sea correcto.") return puntos else: print(f"[ERROR] ❌ Error {response.status_code}: {response.text[:200]}") return puntos except requests.exceptions.Timeout: print("[ERROR] ❌ Tiempo de espera agotado. El archivo de 70 MB tarda en descargarse.") return puntos except Exception as e: print(f"[ERROR] ❌ No se pudo cargar el archivo de ubicaciones: {e}") return puntos # 3. Verificar columnas disponibles print(f"[INFO] Columnas disponibles en ubicaciones: {list(gdf_ubicaciones.columns)}") # Usar NAME_1 como región if 'NAME_1' not in gdf_ubicaciones.columns: print("[ERROR] ❌ No se encontró NAME_1 en el archivo de ubicaciones") return puntos # Buscar campo de localidad locality_field = None if 'NAME_2' in gdf_ubicaciones.columns: locality_field = 'NAME_2' print(f"[INFO] Usando NAME_2 como campo de localidad") elif 'locality' in gdf_ubicaciones.columns: locality_field = 'locality' print(f"[INFO] Usando locality como campo de localidad") elif 'localidad' in gdf_ubicaciones.columns: locality_field = 'localidad' print(f"[INFO] Usando localidad como campo de localidad") else: print("[ADVERTENCIA] ⚠️ No se encontró campo de localidad. Usando NAME_1 como localidad.") gdf_ubicaciones['locality'] = gdf_ubicaciones['NAME_1'] locality_field = 'locality' # 4. Preparar columnas para la unión espacial columnas_union = ['geometry', 'NAME_1'] if locality_field and locality_field != 'locality': columnas_union.append(locality_field) if 'locality' not in gdf_ubicaciones.columns: gdf_ubicaciones['locality'] = gdf_ubicaciones[locality_field] elif locality_field == 'locality': columnas_union.append('locality') print(f"[INFO] Realizando unión espacial con {len(gdf_ubicaciones)} polígonos...") print(f"[INFO] Columnas a unir: {columnas_union}") # 5. Unión espacial try: gdf_con_ubicacion = gpd.sjoin( gdf_puntos, gdf_ubicaciones[columnas_union], how='left', predicate='within' ) print(f"[INFO] ✅ Unión espacial realizada") except Exception as e: print(f"[ERROR] ❌ Error en la unión espacial: {e}") return puntos # 6. Verificar columnas después de la unión print(f"[INFO] Columnas después de la unión: {list(gdf_con_ubicacion.columns)}") # Asegurar que existen las columnas if 'NAME_1' not in gdf_con_ubicacion.columns: gdf_con_ubicacion['NAME_1'] = '' if 'locality' not in gdf_con_ubicacion.columns: if 'NAME_2' in gdf_con_ubicacion.columns: gdf_con_ubicacion['locality'] = gdf_con_ubicacion['NAME_2'] else: gdf_con_ubicacion['locality'] = '' # 7. Rellenar valores nulos gdf_con_ubicacion['NAME_1'] = gdf_con_ubicacion['NAME_1'].fillna('') gdf_con_ubicacion['locality'] = gdf_con_ubicacion['locality'].fillna('') # 8. Reconstruir GeoJSON features = [] for _, row in gdf_con_ubicacion.iterrows(): coords = list(row.geometry.coords)[0] feature = { "type": "Feature", "geometry": { "type": "Point", "coordinates": [coords[0], coords[1]] }, "properties": { 'frp': row.get('frp', 0), 'acq_date': row.get('acq_date', ''), 'acq_time': row.get('acq_time', ''), 'satellite': row.get('satellite', ''), 'brightness': row.get('brightness', None), 'NAME_1': row.get('NAME_1', ''), 'locality': row.get('locality', '') } } features.append(feature) print(f"[INFO] ✅ Total de puntos enriquecidos: {len(features)}") return features def obtener_datos_firms(satellite, api_key): """ Obtiene datos de FIRMS para un satélite específico (últimas 48 horas) """ base_url = "https://firms.modaps.eosdis.nasa.gov/api/area/csv/" bbox = "22,44,41,53" # Ucrania satellites_validos = ['VIIRS_NOAA21_NRT', 'VIIRS_NOAA20_NRT', 'VIIRS_SNPP_NRT'] if satellite not in satellites_validos: print(f"[ERROR] Satélite no soportado: {satellite}") return [] url = f"{base_url}{api_key}/{satellite}/{bbox}/2" headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36" } try: print(f"[INFO] Conectando a FIRMS para: {satellite}...") print(f"[DEBUG] URL: {url}") response = requests.get(url, headers=headers, timeout=60) if response.status_code == 200: content = response.text.strip() print(f"[DEBUG] Longitud del contenido: {len(content)} caracteres") print(f"[DEBUG] Primeros 200 caracteres: {content[:200]}") if not content: print(f"[ADVERTENCIA] No hay datos para {satellite}") return [] lines = content.split('\n') print(f"[DEBUG] Número de líneas: {len(lines)}") if len(lines) < 2: print(f"[ADVERTENCIA] Solo hay cabecera, sin datos para {satellite}") return [] headers_csv = lines[0].split(',') col_map = {} for i, col in enumerate(headers_csv): col_clean = col.strip().lower() if 'latitude' in col_clean: col_map['latitude'] = i elif 'longitude' in col_clean: col_map['longitude'] = i elif 'frp' in col_clean: col_map['frp'] = i elif 'acq_date' in col_clean: col_map['acq_date'] = i elif 'acq_time' in col_clean: col_map['acq_time'] = i elif 'brightness' in col_clean: col_map['brightness'] = i features = [] for line in lines[1:]: if not line.strip(): continue values = line.split(',') try: if 'latitude' in col_map and 'longitude' in col_map and 'frp' in col_map: lat = float(values[col_map['latitude']].strip()) lon = float(values[col_map['longitude']].strip()) frp = float(values[col_map['frp']].strip()) acq_date = values[col_map['acq_date']].strip() if 'acq_date' in col_map else '' acq_time = values[col_map['acq_time']].strip() if 'acq_time' in col_map else '' brightness = float(values[col_map['brightness']].strip()) if 'brightness' in col_map else None feature = { "type": "Feature", "geometry": { "type": "Point", "coordinates": [lon, lat] }, "properties": { "frp": frp, "acq_date": acq_date, "acq_time": acq_time, "satellite": satellite, "brightness": brightness } } features.append(feature) except (ValueError, IndexError): continue print(f"[ÉXITO] {satellite}: {len(features)} puntos detectados") return features else: print(f"[ERROR] Código {response.status_code} para {satellite}") print(f"[DEBUG] Respuesta: {response.text[:200]}") return [] except Exception as e: print(f"[ERROR] Fallo en {satellite}: {e}") return [] def subir_github(archivo, token_git, repo, branch='main'): """ Sube un archivo a GitHub """ if not token_git: print("[ERROR] Token de GitHub no configurado correctamente") return False url_git = f"https://api.github.com/repos/{repo}/contents/{archivo}" headers_git = { "Authorization": f"token {token_git}", "Accept": "application/vnd.github.v3+json" } try: with open(archivo, "rb") as f: contenido_b64 = base64.b64encode(f.read()).decode("utf-8") except FileNotFoundError: print(f"[ERROR] Archivo {archivo} no encontrado") return False sha = None try: response = requests.get(url_git, headers=headers_git, timeout=15) if response.status_code == 200: sha = response.json().get("sha") except Exception as e: print(f"[INFO] No se pudo obtener SHA: {e}") payload = { "message": f"Update {archivo} - {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}", "content": contenido_b64, "branch": branch } if sha: payload["sha"] = sha try: response = requests.put(url_git, json=payload, headers=headers_git, timeout=30) if response.status_code in [200, 201]: print(f"🚀 ¡Archivo {archivo} subido con éxito a GitHub!") return True else: print(f"[ERROR] GitHub devolvió código {response.status_code}") print(f"[DEBUG] Respuesta: {response.text}") return False except Exception as e: print(f"[ERROR] Error de conexión con GitHub: {e}") return False def main(): """ Función principal que orquesta la obtención de datos y subida a GitHub """ FIRMS_API_KEY = os.environ.get("FIRMS_API_KEY") GITHUB_TOKEN = os.environ.get("TOKEN_GITHUB") HF_TOKEN = os.environ.get("HF_TOKEN") GITHUB_REPO = "jaspeado/firms-ukraine-tracker" UBICACIONES_URL = "https://huggingface.co/spaces/Paspis/tracker-nasa/resolve/main/ubicaciones.geojson" if not FIRMS_API_KEY: print("[ERROR] FIRMS_API_KEY no configurada en variables de entorno") print("[INFO] Configúrala en Hugging Face: Settings → Repository secrets") exit(1) if not GITHUB_TOKEN: print("[ERROR] TOKEN_GITHUB no configurado en variables de entorno") print("[INFO] Configúralo en Hugging Face: Settings → Repository secrets") exit(1) if not HF_TOKEN: print("[ADVERTENCIA] ⚠️ HF_TOKEN no configurado. Si el Space es privado, la descarga fallará.") print("[INFO] Configúralo en Hugging Face: Settings → Repository secrets") print(f"[INFO] Obteniendo datos de incendios de FIRMS (últimas 48 horas)") print(f"[INFO] Hora actual: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}") satellites = ['VIIRS_NOAA21_NRT', 'VIIRS_NOAA20_NRT', 'VIIRS_SNPP_NRT'] todos_los_puntos = [] for sat in satellites: puntos = obtener_datos_firms(sat, FIRMS_API_KEY) todos_los_puntos.extend(puntos) time.sleep(1) print(f"[INFO] Enriqueciendo puntos con ubicación...") puntos_enriquecidos = enriquecer_con_ubicacion(todos_los_puntos, UBICACIONES_URL, HF_TOKEN) geojson_final = { "type": "FeatureCollection", "features": puntos_enriquecidos } archivo_salida = "fires.geojson" with open(archivo_salida, "w") as f: json.dump(geojson_final, f, indent=2) print("\n" + "="*50) print(f"RESUMEN DE DATOS (últimas 48h):") print(f" • Total de puntos: {len(puntos_enriquecidos)}") if puntos_enriquecidos: fechas = set() satelites = set() frp_total = 0 ubicados = 0 for feature in puntos_enriquecidos: if 'acq_date' in feature['properties']: fechas.add(feature['properties']['acq_date']) if 'satellite' in feature['properties']: satelites.add(feature['properties']['satellite']) if 'frp' in feature['properties']: frp_total += feature['properties']['frp'] if feature['properties'].get('NAME_1') or feature['properties'].get('locality'): ubicados += 1 print(f" • Fechas: {', '.join(sorted(fechas))}") print(f" • Satélites: {', '.join(satelites)}") print(f" • FRP Total: {frp_total:.2f}") if len(puntos_enriquecidos) > 0: print(f" • FRP Promedio: {frp_total/len(puntos_enriquecidos):.2f}") print(f" • Puntos con ubicación: {ubicados}/{len(puntos_enriquecidos)}") else: print(" • No se detectaron incendios en las últimas 48 horas") print(" • Sugerencia: Prueba con 'world' o amplía el período") print("="*50 + "\n") print(f"[OK] Archivo generado con éxito: {archivo_salida}") subir_github(archivo_salida, GITHUB_TOKEN, GITHUB_REPO, branch='main') if __name__ == "__main__": main()
